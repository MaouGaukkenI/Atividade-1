$ git init
// iniciando o git

$ git add .
//Adicionei todos arquivos que possuía ao git

$ git status
//Visualizei de todos arquivos foram enviados corretamente.

$ git commit -m "Projete inicial"
//Criei um commit marcando o inicio do projeto

$ git status
//visualizei se todas as alterações foram enviadas com sucesso ao commit

$ git remote add origin https://github.com/MaouGaukkenI/Atividade-1.git
//Defini um novo repositório remoto e chamei de "origin" ele tem acesso ao URL informado

$ git branch -m master
//Renomeei a filial local para 'master' para facilitar o código

$ git push -u origin master
//Enviei os commits da branch local "master" para a branch "master" no repositório remoto "origin" e defini "origin/master" como upstream para a branch local "master".

$ git pull origin master
//Utilizei este comando para trazer o arquivo README.md Criado no repositório remoto
